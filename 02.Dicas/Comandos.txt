create a new repository on the command line

echo "# Teste-repositorio-novo" >> README.md
git init
git add README.md
git commit -m "primeiro comentario"
git branch -M main
git remote add origin https://github.com/FierceFox/Teste-repositorio-novo.git
git push -u origin main

push an existing repository from the command line

git remote add origin https://github.com/FierceFox/Teste-repositorio-novo.git
git branch -M main
git push -u origin main
